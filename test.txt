$(document).on("click", ".btnLogin", function() {
	var isMember = $(this).data("isMember");

    // 어린이 시험의 비회원 접수시만 간편 로그인으로 이동. 그 외의 케이스는 모두 next() 에서 처리 한다.
    if (isMember !== "member") {
		var examCate = $("#receiptFrm [name='examCate']").val()

		if (examCate=="JET" || examCate=="JSW" || examCate=="TOB" || examCate=="TBS") {
			$("#receiptFrm").attr("action", "/common/certifyResponse.php");
			window.open("https://www.toeicbridge.co.kr/common/cert/sns/snsLogin.php?frmName=receiptFrm&domain=exam.toeic.co.kr", "_snsLogin", "width=800,height=650 , scrollbars=1");

			return;
		}
	}

    nextWhenNotLogined(isMember === 'member');
});

function nextWhenLogined() {
    // 로그인 되어 있을 경우 step 2 로 이동
    isLogined(function (isLogined) {
        if (isLogined) {
            $("#receiptFrm").attr("action", "/common/certifyResponse.php");
            $("#receiptFrm").submit();
        } else {
            alert('세션이 만료되었습니다. 다시 시도해주시기 바랍니다.');
            location.reload();
        }
    });
}

function nextWhenNotLogined(isMember) {
    if (!isMember) {
        // 비회원 로그인시도일 경우 certifyResponse 로 보내면, 비회원 인증 화면으로 간다.
        $("#receiptFrm").attr("action", "/common/certifyResponse.php");
        $("#receiptFrm").submit();
    } else {
        // family 로그인 시도시 공통 로그인을 통해서 step2로 이동
        // receiptFrm form 에 있는 파라미터를 get방식으로 보내기 위해(family 로그인에선 post방식 처리 불가)
        /* (필수파라미터) 처리 완료 후 이동할 url */
        var returnUrl = $("#receiptFrm [name='returnUrl']").val();

        /* returnUrl 으로 담아서 가져가야할 파라미터 들 */
        var examCate = $("#receiptFrm [name='examCate']").val();
        var examCode = $("#receiptFrm [name='examCode']").val();
        var goodsCode = $("#receiptFrm [name='goodsCode']").val();
        var packGoodsCode = $("#receiptFrm [name='packGoodsCode']").val();
        var examGoodsCode = $("#receiptFrm [name='examGoodsCode']").val();
        var sbGoodsType1 = $("#receiptFrm [name='sbGoodsType1']").val();
        var compExam = $("#receiptFrm [name='compExam']").val();

        /* YBM Family 공통 로그인에선 POST방식 처리불가로 인해 GET방식으로 이동 */
        var url = encodeURIComponent("/common/certifyResponse.php?examCate=" + examCate + "&examCode=" + examCode + "&goodsCode=" + goodsCode + "&packGoodsCode=" + packGoodsCode + "&examGoodsCode=" + examGoodsCode + "&sbGoodsType1=" + sbGoodsType1 + "&returnUrl=" + returnUrl + "&compExam=" + compExam);
        var what = "exam.toeic.co.kr";

        location.href="https://www.ybmnet.co.kr/common/login.asp?url="+ url +"&what=" + what;
    }
}

/* 시험 회원/비회원 로그인 레이어 생성 후 open */
function openExamLayer(isPack) {
	$(".btnLogin").show();

    var goodsName = "";

	// 패키지일 경우 비회원 로그인, 시험날짜 숨김
	if (isPack) {
		goodsName = $("#receiptFrm [name='goodsName']").val();
		$(".btnNoMemberLogin").hide();
		$(".pack").hide();
	// 단일시험일 경우 시험날짜 노출
	} else {
		if ($("#receiptFrm [name='examCate']").val() === "JPN") {
			goodsName = $("#receiptFrm [name='dispGoodsName']").val();
		} else {
			goodsName = fnGetGoodsType1Name($("#receiptFrm [name='sbGoodsType1']").val());
		}

		$(".pack").show();
	}
	$("#layerGoodsName").html(goodsName);

	if ($("#compExam").val() == "1") $(".pack").hide();

	// 시험날짜가 토요일이면 파란글자, 일요일이면 빨간글자로 요일 표기
	if (fnGetDayOfTheWeek($("#receiptFrm [name='examDay']").val())=="토") {
		var examDayColor = "color_02";
	} else if (fnGetDayOfTheWeek($("#receiptFrm [name='examDay']").val())=="일") {
		var examDayColor = "color_01";
	}

	var examCate = $("#receiptFrm [name='examCate']").val();
	var examStartTime = "";
	// CBT시험만 시험시간 표기
	if (examCate!="TOS" && examCate!="JSW" && examCate!="TBS" && examCate!="TSC" && examCate!="SJP") {
		examStartTime = $("#receiptFrm [name='examStartTime']").val();
	}

	$("#layerExamDay").html(fnGetDate($("#receiptFrm [name='examDay']").val(), "-") + "<span class='"+ examDayColor +"'> ("+ fnGetDayOfTheWeek($("#receiptFrm [name='examDay']").val()) +") </span>" + examStartTime);

	$(".layer").show();
}

/* 시험 회원/비회원 로그인 레이어 close */
function closeExamLayer() {
	$(".receiptInfo").val("");
	$(".layer").hide();
}
